package humanPackage;

public class HumanClass {
	private String Name;
	private int Age;
	
	public HumanClass() {
		Age=0;
		System.out.println("New Human!\n");
	}

	public String getName() {
		return Name;
	}
	public void setName(String onoma) {
		Name=onoma;
	}
	public int getAge() {
		return Age;
	}
	public void setAge(int ilikia) {
		Age=ilikia;
	}
  
	  
  }
